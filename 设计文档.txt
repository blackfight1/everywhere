  Hidden Attack Surface Scanner — 完整设计文档（Go + Interactsh + Docker）                                                           
                                                                                                                                     
  ---                                                                                                                                
  一、项目目标                                                                                                                       
                                                                                                                                     
  构建一个 Go 语言编写的独立 Web 应用，自动化实现 Burp Suite 插件 "Collaborator Everywhere v2" 的全部功能，使用                      
  projectdiscovery/interactsh 作为 OOB 回调基础设施（支持自建或使用公共服务器），并深度结合 James Kettle "Cracking the Lens"         
  研究中的高级攻击技术进行增强。无需 Burp Suite。使用 Docker Compose 一键部署。

  五、核心模块详细设计                                                                                                               
                                                                                                                                     
  模块 1：Interactsh 集成层（pkg/oob/）                                                                                              
                                                                                                                                     
  直接使用 interactsh Go client 库，无需自建 DNS/HTTP 服务。                                                                         
                                                                                                                                     
  // pkg/oob/client.go                                                                                                               
                                                                                                                                     
  type OOBClient struct {                                                                                                            
      client       *interactsh_client.Client                                                                                         
      correlations sync.Map  // uniqueID → *CorrelationEntry                                                                         
      ownIPs       sync.Map  // string → bool，扫描器自身出口 IP                                                                     
  }                                                                                                                                  
                                                                                                                                     
  type CorrelationEntry struct {                                                                                                     
      ScanTaskID  string                                                                                                             
      TargetURL   string                                                                                                             
      PayloadType string    // "header" / "param" / "raw"                                                                            
      PayloadKey  string                                                                                                             
      PayloadVal  string                                                                                                             
      SentAt      time.Time                                                                                                          
      FullOOBURL  string                                                                                                             
  }                                                                                                                                  
                                                                                                                                     
  // 创建客户端                                                                                                                      
  func NewOOBClient(serverURL string, token string) (*OOBClient, error) {                                                            
      opts := &interactsh_client.Options{                                                                                            
          ServerURL: serverURL,  // 自建: "oob.yourdomain.com"                                                                       
                                 // 公共: "" 空值使用默认 "oast.pro,oast.live,oast.site,..."                                         
          Token:     token,                                                                                                          
      }                                                                                                                              
      client, err := interactsh_client.New(opts)                                                                                     
      // ...                                                                                                                         
  }                                                                                                                                  
                                                                                                                                     
  // 生成唯一 OOB URL 并注册关联                                                                                                     
  func (o *OOBClient) GeneratePayload(entry CorrelationEntry) string {                                                               
      url := o.client.URL()  // 返回 "ck3r5si1234abcde.oast.pro"                                                                     
      uniqueID := strings.Split(url, ".")[0]                                                                                         
      o.correlations.Store(uniqueID, &entry)                                                                                         
      return url                                                                                                                     
  }                                                                                                                                  
                                                                                                                                     
  // 启动轮询                                                                                                                        
  func (o *OOBClient) StartPolling(callback func(*Pingback)) {                                                                       
      o.client.StartPolling(5*time.Second, func(interaction *server.Interaction) {                                                   
          entry, ok := o.correlations.Load(interaction.UniqueID)                                                                     
          if !ok {                                                                                                                   
              // 检查是否为自身 IP 检测的回调                                                                                        
              o.checkSelfIP(interaction)                                                                                             
              return                                                                                                                 
          }                                                                                                                          
          cor := entry.(*CorrelationEntry)                                                                                           
          fromOwnIP := o.isSelfIP(interaction.RemoteAddress)                                                                         
          pingback := &Pingback{                                                                                                     
              Interaction: interaction,                                                                                              
              Correlation: cor,                                                                                                      
              FromOwnIP:   fromOwnIP,                                                                                                
          }                                                                                                                          
          callback(pingback)                                                                                                         
      })                                                                                                                             
  }                                                                                                                                  
                                                                                                                                     
  // 自身 IP 检测：向 OOB URL 发一个 HTTP 请求，记录出口 IP                                                                          
  func (o *OOBClient) DetectSelfIP() {                                                                                               
      url := o.client.URL()                                                                                                          
      uniqueID := strings.Split(url, ".")[0]                                                                                         
      o.correlations.Store(uniqueID, &CorrelationEntry{PayloadKey: "__self_ip_check__"})                                             
      http.Get("https://" + url)                                                                                                     
  }                                                                                                                                  
                                                                                                                                     
  func (o *OOBClient) checkSelfIP(interaction *server.Interaction) {                                                                 
      entry, ok := o.correlations.Load(interaction.UniqueID)                                                                         
      if ok && entry.(*CorrelationEntry).PayloadKey == "__self_ip_check__" {                                                         
          ip, _, _ := net.SplitHostPort(interaction.RemoteAddress)                                                                   
          if ip == "" { ip = interaction.RemoteAddress }                                                                             
          o.ownIPs.Store(ip, true)                                                                                                   
      }                                                                                                                              
  }                                                                                                                                  
                                                                                                                                     
  func (o *OOBClient) Shutdown() {                                                                                                   
      o.client.StopPolling()                                                                                                         
      o.client.Close()                                                                                                               
  }

  2.2 占位符体系（兼容 Burp 插件格式）                                                                                               
                                                                                                                                     
  ┌────────┬─────────────────────────────┬───────────────────────────────────┐                                                       
  │ 占位符 │           替换为            │               说明                │                                                       
  ├────────┼─────────────────────────────┼───────────────────────────────────┤                                                       
  │ %s     │ client.URL() 返回值         │ 每个 payload 实例独立 URL         │                                                       
  ├────────┼─────────────────────────────┼───────────────────────────────────┤                                                       
  │ %h     │ 目标请求的 Host 值          │ 用于 Host header 变体             │                                                       
  ├────────┼─────────────────────────────┼───────────────────────────────────┤                                                       
  │ %o     │ 扫描配置中提供的 Origin 值  │ 仅配置了 Origin 时启用，否则跳过  │                                                       
  ├────────┼─────────────────────────────┼───────────────────────────────────┤                                                       
  │ %r     │ 扫描配置中提供的 Referer 值 │ 仅配置了 Referer 时启用，否则跳过 │                                                       
  └────────┴─────────────────────────────┴───────────────────────────────────┘                                                       
                                                                                                                                     
  注意：Burp 插件是代理模式，能从真实浏览器请求中获取 Origin/Referer。我们是主动扫描模式，需要用户在扫描配置 custom_headers 中提供   
  Origin/Referer，或者自动用 https://{target_host}/ 作为 Referer 默认值。

  3.2 发送策略                                                                                                                       
                                                                                                                                     
  策略（与 Burp 插件一致）：对每个 target 发送 1 个请求，该请求同时携带所有 active payload header（每个 header 使用不同的 OOB        
  URL）。回调通过不同的 UniqueID 精确定位是哪个 header 触发的。                                                                      
                                                                                                                                     
  对于 raw 类型 payload（Cracking the Lens），每个 raw payload 必须单独发一个请求，因为它们修改的是请求行或 TLS 层，无法合并。       
                                                                                                                                     
  3.3 标准 HTTP 发送器                                                                                                               
                                                                                                                                     
  // pkg/scanner/http_sender.go                                                                                                      
                                                                                                                                     
  func SendStandardRequest(ctx context.Context, target string, payloads []ResolvedPayload, client *http.Client) (*http.Response,     
  error) {                                                                                                                           
      req, _ := http.NewRequestWithContext(ctx, "GET", target, nil)                                                                  
      req.Header.Set("Cache-Control", "no-transform")                                                                                
                                                                                                                                     
      for _, p := range payloads {                                                                                                   
          switch p.Type {                                                                                                            
          case PayloadTypeHeader:                                                                                                    
              req.Header.Set(p.Key, p.Value)                                                                                         
          case PayloadTypeParam:                                                                                                     
              q := req.URL.Query()                                                                                                   
              q.Set(p.Key, p.Value)                                                                                                  
              req.URL.RawQuery = q.Encode()                                                                                          
          }                                                                                                                          
      }                                                                                                                              
                                                                                                                                     
      return client.Do(req)                                                                                                          
  }                                                                                                                                  
                                                                                                                                     
  3.4 Raw Socket 发送器                                                                                                              
                                                                                                                                     
  // pkg/scanner/raw_sender.go                                                                                                       
                                                                                                                                     
  type RawRequest struct {                                                                                                           
      Host     string                                                                                                                
      Port     int                                                                                                                   
      UseTLS   bool                                                                                                                  
      SNIHost  string   // TLS SNI hostname，可与 Host header 不同                                                                   
      RawBytes []byte   // 完整的原始 HTTP 请求字节                                                                                  
  }                                                                                                                                  
                                                                                                                                     
  func SendRawRequest(ctx context.Context, r RawRequest) ([]byte, error) {                                                           
      dialer := &net.Dialer{Timeout: 10 * time.Second}                                                                               
      conn, err := dialer.DialContext(ctx, "tcp", fmt.Sprintf("%s:%d", r.Host, r.Port))                                              
      if err != nil {                                                                                                                
          return nil, err                                                                                                            
      }                                                                                                                              
      defer conn.Close()                                                                                                             
                                                                                                                                     
      if r.UseTLS {                                                                                                                  
          tlsConn := tls.Client(conn, &tls.Config{                                                                                   
              InsecureSkipVerify: true,                                                                                              
              ServerName:         r.SNIHost,                                                                                         
          })                                                                                                                         
          if err := tlsConn.HandshakeContext(ctx); err != nil {                                                                      
              return nil, err                                                                                                        
          }                                                                                                                          
          conn = tlsConn                                                                                                             
      }                                                                                                                              
                                                                                                                                     
      conn.SetDeadline(time.Now().Add(10 * time.Second))                                                                             
      conn.Write(r.RawBytes)                                                                                                         
                                                                                                                                     
      buf := make([]byte, 65535)                                                                                                     
      n, _ := conn.Read(buf)                                                                                                         
      return buf[:n], nil                                                                                                            
  }                                                                                                                                  
                                                                                                                                     
  3.5 Cracking the Lens 请求构建器                                                                                                   
                                                                                                                                     
  // pkg/scanner/cracking.go                                                                                                         
                                                                                                                                     
  func BuildCrackingRequests(targetURL string, p Payload, oobURL string) []RawRequest {                                              
      parsed, _ := url.Parse(targetURL)                                                                                              
      host := parsed.Hostname()                                                                                                      
      port := resolvePort(parsed) // 默认 https=443, http=80                                                                         
      path := parsed.Path                                                                                                            
      if path == "" { path = "/" }                                                                                                   
      useTLS := parsed.Scheme == "https"                                                                                             
                                                                                                                                     
      var requests []RawRequest                                                                                                      
                                                                                                                                     
      switch p.Key {                                                                                                                 
      case "absolute-url-host-mismatch":                                                                                             
          raw := fmt.Sprintf("GET https://%s%s HTTP/1.1\r\nHost: %s\r\nConnection: close\r\n\r\n", oobURL, path, host)               
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "duplicate-host":                                                                                                         
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s\r\nHost: %s\r\nConnection: close\r\n\r\n", path, host, oobURL)             
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "host-with-at":                                                                                                           
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s@%s\r\nConnection: close\r\n\r\n", path, oobURL, host)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "host-at-reversed":                                                                                                       
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s@%s\r\nConnection: close\r\n\r\n", path, host, oobURL)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "host-with-hash":                                                                                                         
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s#%s\r\nConnection: close\r\n\r\n", path, host, oobURL)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "host-crlf-inject":                                                                                                       
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s\r\nX-Injected: %s\r\nConnection: close\r\n\r\n", path, host, oobURL)       
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "host-with-space":                                                                                                        
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s %s\r\nConnection: close\r\n\r\n", path, host, oobURL)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "path-at-prefix":                                                                                                         
          raw := fmt.Sprintf("GET @%s%s HTTP/1.1\r\nHost: %s\r\nConnection: close\r\n\r\n", oobURL, path, host)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "path-slash-prefix":                                                                                                      
          raw := fmt.Sprintf("GET /%s%s HTTP/1.1\r\nHost: %s\r\nConnection: close\r\n\r\n", oobURL, path, host)                      
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: useTLS, SNIHost: host, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "sni-host-mismatch":                                                                                                      
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s\r\nConnection: close\r\n\r\n", path, host)                                 
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: true, SNIHost: oobURL, RawBytes: []byte(raw)})      
                                                                                                                                     
      case "sni-host-mismatch-reversed":                                                                                             
          raw := fmt.Sprintf("GET %s HTTP/1.1\r\nHost: %s\r\nConnection: close\r\n\r\n", path, oobURL)                               
          requests = append(requests, RawRequest{Host: host, Port: port, UseTLS: true, SNIHost: host, RawBytes: []byte(raw)})        
                                                                                                                                     
      case "alt-ports":                                                                                                              
          // 由扫描引擎上层处理：对 8080,8443,4443,8888,8000,9090,9443 重复全部标准 payload                                          
      }                                                                                                                              
                                                                                                                                     
      return requests                                                                                                                
  }

  }                                                                                                                                  
                                                                                                                                     
  // pkg/correlator/severity.go                                                                                                      
                                                                                                                                     
  func EvaluateSeverity(protocol string, fromOwnIP bool) string {                                                                    
      if fromOwnIP {                                                                                                                 
          return "low"                                                                                                               
      }                                                                                                                              
      switch protocol {                                                                                                              
      case "dns":                                                                                                                    
          return "medium"     // 后端解析了域名，不一定发起了请求                                                                    
      case "http", "https":                                                                                                          
          return "high"       // 后端发起了 HTTP 请求 → SSRF                                                                         
      case "smtp":                                                                                                                   
          return "high"       // 后端发了邮件                                                                                        
      case "ldap":                                                                                                                   
          return "critical"   // LDAP 外带 → Log4Shell 类                                                                            
      case "ftp":                                                                                                                    
          return "high"       // FTP 外带                                                                                            
      default:                                                                                                                       
          return "medium"                                                                                                            
      }                                                                                                                              
  }

  八、关键实现注意事项                                                                                                               
                                                                                                                                     
  1. 每个 payload 的 OOB URL 必须唯一 — 每次调用 client.URL() 生成新 URL，确保回调能精确关联到具体的 target + payload 组合           
  2. Forwarded header 特殊处理 — 值中有 3 个 %s，全部替换为同一个 OOB URL（只调用一次 client.URL()）                                 
  3. %o 和 %r 占位符 — 主动扫描无浏览器上下文，需使用扫描配置中的 default_origin/default_referer，若未配置则跳过对应 payload         
  4. Cache-Control: no-transform — 所有请求固定携带，防止中间代理修改内容                                                            
  5. interactsh client 生命周期 — 每个扫描任务独立实例，完成后 StopPolling → Close 正确注销                                          
  6. Raw 请求用 net.Conn — Go net/http 不支持畸形请求行/双 Host/CRLF 注入，必须直接操作 TCP                                          
  7. 去重 — 同一个 UniqueID 可能触发多次 DNS 查询，按 UniqueID 去重，只记录第一次回调                                                
  8. interactsh 公共服务器 — 不填 INTERACTSH_SERVER 时默认使用
